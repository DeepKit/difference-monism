## 四、快速评估：零成本判断复杂度

不用调用LLM，纯规则就能判断大部分情况：

### 因子1：关键词检测

```python
COMPLEXITY_KEYWORDS = {
    "简单": ["翻译", "总结", "是什么", "定义", "列出"],
    "中等": ["分析", "比较", "解释", "为什么"],
    "复杂": ["设计", "实现", "优化", "重构"],
    "专家": ["证明", "推导", "分布式", "架构"]
}

def detect_by_keywords(text):
    for level, keywords in COMPLEXITY_KEYWORDS.items():
        if any(kw in text for kw in keywords):
            return level
    return "中等"  # 默认
```

### 因子2：消息长度

```python
def detect_by_length(messages):
    total_length = sum(len(m.content) for m in messages)
    
    if total_length < 100:
        return "简单"
    elif total_length < 500:
        return "中等"
    elif total_length < 2000:
        return "复杂"
    else:
        return "专家"
```

### 因子3：历史相似度

```python
def detect_by_history(request):
    # 找到相似的历史请求
    similar = find_similar_requests(request, top_k=5)
    
    if similar:
        # 用历史数据的平均复杂度
        return average([s.actual_complexity for s in similar])
    
    return None  # 无历史数据
```

### 综合评分

```python
def fast_evaluate(request):
    scores = []
    confidences = []
    
    # 关键词（权重30%）
    kw_result = detect_by_keywords(request.content)
    scores.append(LEVEL_TO_SCORE[kw_result] * 0.3)
    confidences.append(0.7)
    
    # 长度（权重20%）
    len_result = detect_by_length(request.messages)
    scores.append(LEVEL_TO_SCORE[len_result] * 0.2)
    confidences.append(0.6)
    
    # 历史（权重30%）
    hist_result = detect_by_history(request)
    if hist_result:
        scores.append(hist_result * 0.3)
        confidences.append(0.85)
    
    # 领域（权重20%）
    domain_result = detect_domain(request)
    scores.append(DOMAIN_COMPLEXITY[domain_result] * 0.2)
    confidences.append(0.65)
    
    return {
        "complexity": sum(scores),
        "confidence": average(confidences)
    }
```

## 五、精确评估：LLM辅助判断

当快速评估置信度<70%时，花一点小钱让LLM帮忙判断：

```python
EVAL_PROMPT = """
分析这个任务的复杂度（1-10分）：

任务：{task}

评估维度：
- 推理深度：需要多少步推理？
- 领域专业性：需要专业知识吗？
- 输出复杂度：输出有多复杂？
- 歧义程度：任务描述清晰吗？

返回JSON：{"score": 1-10, "reason": "简短理由"}
"""

def precise_evaluate(request):
    response = call_llm(
        model="Basic Model-turbo",  # 用便宜模型评估
        prompt=EVAL_PROMPT.format(task=request.content),
        max_tokens=100
    )
    return json.loads(response)["score"]
```

**成本**：约$0.001/次，但能大幅提高路由准确率。

```python
MODEL_TIERS = {

        "model": "Advanced Model-turbo",
        "cost": 0.01
    }
}

def select_model(complexity, budget_status):
    # 找到对应层级
    for tier, config in MODEL_TIERS.items():
        if config["range"][0] <= complexity <= config["range"][1]:
            # 预算紧张时降级
            if budget_status == "tight" and tier != "budget":
                return get_cheaper_alternative(tier)
            return config["model"]
    
    return "Basic Model-turbo"  # 默认
```

## 八、成本熔断器

防止预算失控：

```python
class CostCircuitBreaker:
    def __init__(self, daily_budget, hourly_limit):
        self.daily_budget = daily_budget
        self.hourly_limit = hourly_limit
    
    def check(self, estimated_cost):
        daily_spent = get_daily_spend()
        hourly_spent = get_hourly_spend()
        
        # 接近日预算，强制降级
        if daily_spent + estimated_cost > self.daily_budget * 0.9:
            return {"action": "downgrade", "model": "Intermediate Model-chat"}
        
        # 超过小时限制，排队等待
        if hourly_spent > self.hourly_limit:
            return {"action": "queue", "retry_after": next_hour()}
        
        return {"action": "proceed"}
```
